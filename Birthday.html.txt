        // ================================================================
        //  CONFETTI
        // ================================================================
        let confettiPieces = [];
        let confettiRunning = false;
        let confettiRAF = null;

        function resizeConfetti() {
            confettiCanvas.width = window.innerWidth;
            confettiCanvas.height = window.innerHeight;
        }
        window.addEventListener('resize', resizeConfetti);
        resizeConfetti();

        class ConfettiPiece {
            constructor() {
                this.x = Math.random() * confettiCanvas.width;
                this.y = Math.random() * confettiCanvas.height - confettiCanvas.height;
                this.w = 6 + Math.random() * 10;
                this.h = 4 + Math.random() * 8;
                this.color = ['#f5a78e', '#d45a3a', '#f5d9cb', '#7bc6a4', '#f5c8b8', '#ffd166', '#ff6b7a', '#c084f5', '#6bc4f5'][
                    Math.floor(Math.random() * 9)
                ];
                this.vx = (Math.random() - 0.5) * 2;
                this.vy = 1.2 + Math.random() * 2.6;
                this.rotation = Math.random() * 360;
                this.rotSpeed = (Math.random() - 0.5) * 8;
                this.opacity = 0.7 + Math.random() * 0.3;
            }
            update() {
                this.x += this.vx + Math.sin(this.y * 0.01) * 0.4;
                this.y += this.vy;
                this.rotation += this.rotSpeed;
                if (this.y > confettiCanvas.height + 30) {
                    this.y = -30;
                    this.x = Math.random() * confettiCanvas.width;
                    this.vy = 1.2 + Math.random() * 2.6;
                }
                if (this.x < -30) this.x = confettiCanvas.width + 30;
                if (this.x > confettiCanvas.width + 30) this.x = -30;
            }
            draw(ctx) {
                ctx.save();
                ctx.translate(this.x, this.y);
                ctx.rotate((this.rotation * Math.PI) / 180);
                ctx.globalAlpha = this.opacity;
                ctx.fillStyle = this.color;
                ctx.shadowColor = 'rgba(0,0,0,0.08)';
                ctx.shadowBlur = 6;
                ctx.fillRect(-this.w / 2, -this.h / 2, this.w, this.h);
                ctx.restore();
            }
        }

        function startConfetti(count = 150) {
            if (confettiRunning) return;
            confettiRunning = true;
            confettiCanvas.classList.add('active');
            confettiPieces = [];
            for (let i = 0; i < count; i++) {
                const p = new ConfettiPiece();
                p.y = Math.random() * confettiCanvas.height;
                confettiPieces.push(p);
            }
            animateConfetti();
        }

        function stopConfetti() {
            confettiRunning = false;
            if (confettiRAF) {
                cancelAnimationFrame(confettiRAF);
                confettiRAF = null;
            }
            confettiCanvas.classList.remove('active');
            ctx.clearRect(0, 0, confettiCanvas.width, confettiCanvas.height);
        }

        function animateConfetti() {
            if (!confettiRunning) return;
            ctx.clearRect(0, 0, confettiCanvas.width, confettiCanvas.height);
            confettiPieces.forEach(p => {
                p.update();
                p.draw(ctx);
            });
            confettiRAF = requestAnimationFrame(animateConfetti);
        }

        // ================================================================
        //  PASSWORD LOGIC
        // ================================================================
        function handlePassword() {
            const val = passwordInput.value.trim();
            if (val === CORRECT_PASSWORD) {
                passwordError.textContent = '';
                passwordInput.value = '';
                showScreen(2);
            } else {
                passwordError.textContent = '❌ Wrong password! Try again Ashiii 💕';
                passwordInput.value = '';
                passwordInput.focus();
                // Shake
                passwordInput.style.animation = 'shake 0.4s ease';
                setTimeout(() => passwordInput.style.animation = '', 500);
            }
        }

        // Add shake keyframes dynamically
        const styleSheet = document.createElement('style');
        styleSheet.textContent = `
                    @keyframes shake {
                        0%, 100% { transform: translateX(0); }
                        20% { transform: translateX(-12px); }
                        40% { transform: translateX(12px); }
                        60% { transform: translateX(-8px); }
                        80% { transform: translateX(8px); }
                    }
                `;
        document.head.appendChild(styleSheet);

        passwordBtn.addEventListener('click', handlePassword);

            if (e.key === 'Enter') handlePassword();
        });
